\begin{defin}
En annnengradslikning er et utrykk på formen:
\begin{displaymath}
a\cdot x^2 + b\cdot x + c = 0.
\end{displaymath}
\end{defin}

\item Dette kan blandt annet relateres til kastebaner og geometriske problemer.\\
\item Grafisk drøfning gjør at vi enkelt kan observere tilfeller der grafen skjærer $x$ aksen:
\end{itemize}

\begin{frame}{Algebraisk løsning}
  \begin{itemize}
  \item Vi angriper det algebraiske problemet med å løse annengradslikninger.
  \item Vi skal benytte oss av en metode som kalles fullstedig kvadrat/ fullføring av kvadrat.
  \item Dette benytter seg av kvadratsetningene.
 \end{itemize}
\begin{repi}[Kvadratsetningene]
\[
(\alpha \pm \beta)^2 = \alpha^2 \pm2\alpha\beta+\beta^2
\]
\end{repi}
\end{frame}


\begin{frame}{Eksempel på fullføring av kvadrat:}
  \begin{align*}
    x^2 + 2x - 8 & = 0 \\
    x^2 + 2x & = 8 \\
  \end{align*}
Legg til en på begge sider:
\begin{align*}
  x^2 + 2x + 1 = 9
\end{align*}
Ved hjelp av første-kvadratsetning ser vi at for $\alpha = x, \beta = 1$ har vi at:
\begin{align*}
  \left( x+1 \right)^2 = 9 
\end{align*}
Trekk så roten på begge sider og flytt over:
\begin{align*}
  x + 1 & = \pm 3\\
  x & = \pm 3 -1
\end{align*}
\end{frame}

\begin{frame}{$abc$-formelen}
  \begin{itemize}
  \item Dette gir elevene en metode å løse likninger på.
  \item Dette motiverer og illustrerer hvordan $abc$-formelen utledes.
  \item Deretter vil elevene arbeide med hvordan man løser andregradslikninger ved hjelp av $abc$-formelen
  \end{itemize}
  \begin{resultat}
    For ett gitt annengradsutrykk
\[
a \cdot x^2 + b\cdot x + c = 0
\]
Vil løsningen kunne utrykkes som
\[
x = \frac{-b \pm \sqrt{b^2 -4ac}}{2a}
\]
  \end{resultat}
\end{frame}
\begin{frame}{Konklusjon}
  \begin{itemize}
  \item Den grafiske tolkningen av annengradslikninger gir ett enkelt bilde av hva det vil si å løse en slik likning og hvordan: flere,unik og ingen løsning kan oppstå. 
  \item Dette gjør det at det blir lett å relatere paraboler til kasterbaner. 
  \item Vi gjør dette for å illustrere at $abc$-formelen ikke er en magisk formel men at det finns ett system bak det.
  \item Disse metodene styrker elevenes forståelse av koblingen mellom grafer og algebra, samnt får repetert og sett nytten av kvadratsetningene.
  \end{itemize}
\end{frame}
